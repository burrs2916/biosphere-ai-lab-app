### 2.2 文件验证

```javascript
// 验证函数
function validateFile(file) {
    const errors = [];
    
    // 1. 检查文件是否存在
    if (!file) {
        errors.push('未选择文件');
        return { valid: false, errors };
    }
    
    // 2. 检查文件类型
    const allowedTypes = ['image/png', 'image/jpeg', 'image/webp'];
    if (!allowedTypes.includes(file.type)) {
        errors.push(`不支持的文件类型: ${file.type}，仅支持 PNG/JPG/WebP`);
    }
    
    // 3. 检查文件大小（最大 10MB）
    const maxSize = 10 * 1024 * 1024; // 10MB
    if (file.size > maxSize) {
        errors.push(`文件过大: ${formatSize(file.size)}，最大支持 10MB`);
    }
    
    // 4. 检查文件名称
    if (file.name.length > 255) {
        errors.push('文件名过长');
    }
    
    return {
        valid: errors.length === 0,
        errors,
        info: {
            name: file.name,
            type: file.type,
            size: file.size,
            lastModified: new Date(file.lastModified)
        }
    };
}

// 文件大小格式化
function formatSize(bytes) {
    if (bytes < 1024) return bytes + ' B';
    if (bytes < 1024 * 1024) return (bytes / 1024).toFixed(1) + ' KB';
    return (bytes / (1024 * 1024)).toFixed(2) + ' MB';
}

// 使用示例
function processFile(file) {
    const validation = validateFile(file);
    
    if (!validation.valid) {
        console.error('文件验证失败:', validation.errors);
        showError(validation.errors.join('\n'));
        return;
    }
    
    console.log('文件验证通过:', validation.info);
    // 继续处理...
}
```

### 2.3 ArrayBuffer 生成

#### 方法一：FileReader 读取为 ArrayBuffer

```javascript
function fileToArrayBuffer(file) {
    return new Promise((resolve, reject) => {
        const reader = new FileReader();
        
        // 读取完成
        reader.onload = function(e) {
            const arrayBuffer = e.target.result;
            console.log('ArrayBuffer 生成成功');
            console.log('  - 字节长度:', arrayBuffer.byteLength);
            console.log('  - 数据类型:', Object.prototype.toString.call(arrayBuffer));
            resolve(arrayBuffer);
        };
        
        // 读取出错
        reader.onerror = function(e) {
            reject(new Error('文件读取失败: ' + e.target.error));
        };
        
        // 开始读取为 ArrayBuffer
        reader.readAsArrayBuffer(file);
    });
}

// 使用示例
async function processFile(file) {
    try {
        const arrayBuffer = await fileToArrayBuffer(file);
        console.log('ArrayBuffer:', {
            byteLength: arrayBuffer.byteLength,
            type: 'ArrayBuffer'
        });
    } catch (error) {
        console.error(error);
    }
}
```

### 2.4 ArrayBuffer 验证

```javascript
// 验证 ArrayBuffer 数据
function validateArrayBuffer(buffer) {
    const errors = [];
    const info = {};
    
    // 1. 检查是否为 ArrayBuffer
    if (!(buffer instanceof ArrayBuffer)) {
        errors.push('数据类型错误，不是 ArrayBuffer');
        return { valid: false, errors };
    }
    
    // 2. 检查字节长度
    const byteLength = buffer.byteLength;
    info.byteLength = byteLength;
    
    if (byteLength === 0) {
        errors.push('数据为空');
    }
    
    // 3. 检查最小长度（PNG 文件头 8 字节）
    if (byteLength < 8) {
        errors.push('数据过小，可能不是有效的图像文件');
    }
    
    // 4. 检查文件头（魔数）
    if (byteLength >= 8) {
        const header = new Uint8Array(buffer, 0, 8);
        info.header = Array.from(header);
        
        // PNG: 89 50 4E 47 0D 0A 1A 0A
        const isPNG = header[0] === 0x89 && header[1] === 0x50;
        
        // JPEG: FF D8 FF
        const isJPEG = header[0] === 0xFF && header[1] === 0xD8 && header[2] === 0xFF;
        
        // WebP: RIFF....WEBP
        const isWEBP = header[0] === 0x52 && header[1] === 0x49 && 
                       header[2] === 0x46 && header[3] === 0x46; // "RIFF"
        
        info.format = isPNG ? 'PNG' : isJPEG ? 'JPEG' : isWEBP ? 'WebP' : 'Unknown';
        
        if (!isPNG && !isJPEG && !isWEBP) {
            errors.push(`无法识别的图像格式，文件头: 0x${header.map(b => b.toString(16).padStart(2, '0')).join(' 0x')}`);
        }
    }
    
    return {
        valid: errors.length === 0,
        errors,
        info
    };
}

// 使用示例
async function processFile(file) {
    const arrayBuffer = await fileToArrayBuffer(file);
    
    // 验证 ArrayBuffer
    const validation = validateArrayBuffer(arrayBuffer);
    
    console.log('ArrayBuffer 验证结果:');
    console.log('  - 有效:', validation.valid);
    console.log('  - 信息:', validation.info);
    
    if (!validation.valid) {
        console.error('  - 错误:', validation.errors);
    }
}
```

```javascript
async function sendToBackend(arrayBuffer, fileName) {
    // 创建 FormData
    const formData = new FormData();
    
    // 将 ArrayBuffer 转为 Blob
    const blob = new Blob([arrayBuffer], { type: 'image/png' });
    formData.append('image', blob, fileName);
    
    // 发送请求
    const response = await fetch('/api/analyze', {
        method: 'POST',
        body: formData
    });
    
    if (!response.ok) {
        throw new Error(`请求失败: ${response.status}`);
    }
    
    const result = await response.json();
    return result;
}
```

```javascript
async function sendBase64ToBackend(dataURL) {
```

```rust
#[derive(Debug, Serialize, Deserialize)]
pub struct ImageInfo {
    pub width: u32,
    pub height: u32,
    pub channels: u32,
    pub format: String,
    pub size_bytes: usize,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct AnalyzeResponse {
    pub success: bool,
    pub info: Option<ImageInfo>,
    pub error: Option<String>,
}

async fn analyze_image(mut multipart: Multipart) -> Result<Json<AnalyzeResponse>, StatusCode> {
    // 获取上传的文件
    let field = multipart.next_field()
        .await
        .map_err(|_| StatusCode::BAD_REQUEST)?
        .ok_or(StatusCode::BAD_REQUEST)?;
    
    let file_name = field.file_name()
        .unwrap_or("unknown")
        .to_string();
    
    // 获取文件内容（Bytes）
    let bytes = field.bytes()
        .await
        .map_err(|_| StatusCode::BAD_REQUEST)?;
    
    println!("[后端] 收到图像: {}", file_name);
    println!("[后端] 数据大小: {} 字节", bytes.len());
    
    // 验证数据
    if bytes.is_empty() {
        return Ok(Json(AnalyzeResponse {
            success: false,
            info: None,
            error: Some("文件为空".to_string()),
        }));
    }
    
    // 调用图像处理函数
    match process_image(&bytes) {
        Ok(info) => {
            println!("[后端] 图像处理成功: {}x{}", info.width, info.height);
            Ok(Json(AnalyzeResponse {
                success: true,
                info: Some(info),
                error: None,
            }))
        }
        Err(e) => {
            println!("[后端] 图像处理失败: {}", e);
            Ok(Json(AnalyzeResponse {
                success: false,
                info: None,
                error: Some(e.to_string()),
            }))
        }
    }
}
```

### 3.2 图像解码模块

```rust
// src/image/loader.rs

use image::{DynamicImage, ImageFormat, ImageReader};
use std::io::Cursor;

#[derive(Debug)]
pub struct ImageData {
    pub width: u32,
    pub height: u32,
    pub format: String,
    pub pixels: Vec<u8>,  // RGBA 格式
}

#[derive(Debug, thiserror::Error)]
pub enum ImageError {
    #[error("无法识别图像格式")]
    UnknownFormat,
    
    #[error("图像解码失败: {0}")]
    DecodeError(String),
    
    #[error("不支持的通道数: {0}")]
    UnsupportedChannels(usize),
}

pub fn load_image(bytes: &[u8]) -> Result<ImageData, ImageError> {
    // 1. 识别格式
    let format = detect_format(bytes)?;
    println!("[loader] 检测到格式: {:?}", format);
    
    // 2. 解码图像
    let img = decode_image(bytes, format)?;
    println!("[loader] 解码完成: {}x{}", img.width(), img.height());
    
    // 3. 转换为 RGBA
    let rgba = img.to_rgba8();
    let (width, height) = rgba.dimensions();
    let pixels = rgba.into_raw();
    
    println!("[loader] 像素数据: {} 字节", pixels.len());
    
    Ok(ImageData {
        width,
        height,
        format: format_to_string(format),
        pixels,
    })
}

fn detect_format(bytes: &[u8]) -> Result<ImageFormat, ImageError> {
    image::guess_format(bytes)
        .map_err(|_| ImageError::UnknownFormat)
}

fn decode_image(bytes: &[u8], format: ImageFormat) -> Result<DynamicImage, ImageError> {
    image::load_from_memory_with_format(bytes, format)
        .map_err(|e| ImageError::DecodeError(e.to_string()))
}

fn format_to_string(format: ImageFormat) -> String {
    match format {
        ImageFormat::Png => "PNG".to_string(),
        ImageFormat::Jpeg => "JPEG".to_string(),
        ImageFormat::WebP => "WebP".to_string(),
        _ => "Unknown".to_string(),
    }
}
```

## 五、错误处理

### 5.1 前端错误

```javascript
// 错误类型
const ErrorTypes = {
    NO_FILE: '未选择文件',
    INVALID_TYPE: '文件类型不支持',
    FILE_TOO_LARGE: '文件过大',
    READ_ERROR: '文件读取失败',
    UPLOAD_ERROR: '上传失败',
    SERVER_ERROR: '服务器错误'
};

// 处理函数
function handleError(error) {
    console.error('[Error]', error);
    
    let message = '发生错误';
    if (error.type === 'NO_FILE') message = '请先选择文件';
    else if (error.type === 'INVALID_TYPE') message = '仅支持 PNG/JPG/WebP 格式';
    else if (error.type === 'FILE_TOO_LARGE') message = '文件大小不能超过 10MB';
    else if (error.type === 'READ_ERROR') message = '文件读取失败，请重试';
    else if (error.type === 'UPLOAD_ERROR') message = '上传失败，请检查网络';
    else if (error.type === 'SERVER_ERROR') message = '服务器错误: ' + error.detail;
    
    showError(message);
}
```

```
        ├─→ FormData + Blob
        │   或
        ├─→ JSON + Base64
        │   或
        └─→ JSON + 字节数组
            │
            ▼
步骤 6: 后端处理
        │
        ├─→ 接收二进制数据
        ├─→ image::guess_format() → 识别格式
        ├─→ image::load_from_memory() → 解码
        ├─→ img.to_rgba8() → 转换 RGBA
        └─→ 返回结果
            │
            ▼
步骤 7: 前端显示结果
        │
        ├─→ 显示图像信息
        ├─→ 显示预览图
        └─→ 准备下一步处理
```
